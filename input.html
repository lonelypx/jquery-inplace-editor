				$(document).ready(function(){
					$('.price').inPlaceEditor({event:'click',callback:onUpdate,beforeEdit:editBigin});
					
				});


				function onUpdate(o){

					$('#info').attr("class","");// remove all class from info div 
					if(isNaN(o.val)){// check for value
					$('#info').text("Please Input a Numeric value");
					$('#info').addClass("error");
					$(o.target).text(lastvalue);// assign the old value
					}else{
					$('#info').text("Text Updated to "+ o.val);
					$('#info').addClass("success");
					}

				}
				
				var lastvalue="";
				function editBigin(o){
					lastvalue=$(o.target).text();
				}
