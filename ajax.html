How to Load data

Click any Item TD to edit. values are loaded form sample.html
NO ITEM QUTY PRICE
1 Item 1 5 100
2 Item 2 5 200
3 Item 3 2 500

Code

				$(document).ready(function(){
					$('.item').inPlaceEditor({event:'click',callback:onUpdate,beforeEdit:editBigin,type:'select',
						provider:{type:'URL',data:'sample.html'}});// init the .item as inplace editable
					
				});

				//provider:{type:'URL',data:'sample.html'} ajax data
				//provider:{type:'URL',data:'<option value="a">a</option>'} embeded data


				function onUpdate(o){

					$('#info').attr("class","");// remove all class from info div 
					$('#info').text("Text Updated to "+ o.val);
					$('#info').addClass("success");

				}
				
				var lastvalue="";
				function editBigin(o){
					lastvalue=$(o.target).text();
				}
